Базовые команды Linux.

Date 01 Ноябрь 2014 Category linux Tags linux

Основные команды

pwd  #текущая директория
ls  #список файлов
man ls #мануал имя команды(ls в данном случае)
clear #очистить экран терминала
exit  #закрыть терминал

mkdir путь #создать папку
touch путь #создать файл
cd 'My Folder' #переходим в папку
cd .. #на 1 уровень вверх
rm -rf #удалить папку со всеми подпапками и файлами внутри
cp путь откуда и путь куда # копируем
cp -r #копировать папку
mv путь1 путь2 #передвигаем
chmod +x nameprog.py #задать права
./progname & #в фоновом режиме или  Ctrl+Z, bg

#записывать вывод из терминала в файл.
program >out.txt 2> file.txt <in.txt (stdout, stder, stdin)

#порядок в аргументах к команде не важен
#ctrl+a , ctrl+e переход в начало или конец строки 
#Tab предложение продолжения команд
#ипользуем nano редактор для редактирования файлов

wget

wget -O  или -P путь_до_директории ссылка # скачивания файлов
--spider # проверить ссылку
-i text.txt #скачать по ссылкам из файла
wget -r -l 1 -A jpg ссылка #скачать все jpg до 1 уровня со страниц

zip

unzip
gunzip   #удаляет архив после распаковки
zip архив.zip файл1 файл2…
tar -zcvf архив.tar файл1 файл2… - создаем архив файлов и запаковываем gunzip
tar -xzvf архив.tar.gz распаковываем

grep

find ~/ -name “*.txt” найти все txt файлы в домашней директории
grep “строка” файл  найти строку в файле
grep -c “srt” file  посчитать кол-во вхождений строки
grep -r “str” folder  найти строку во всех файлах в папке
