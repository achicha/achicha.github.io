Используем Git для контроля версий.

Date 01 Ноябрь 2014 Category web Tags python / git

Установка

sudo apt-get install python-pip git-core

Базовые настройки

git config --global user.name "Your name"
git config --global user.email yourmail@example.com
#инициализируем
git init
#создаем ssh-key для компьютера
ssh-keygen
# копируем содержимое ключа из ssh keys для создания ключа на github
cat ~/.ssh/id_rsa.pub
#клонируем реп с гита # в процессе вводим контрольную фразу 2 раза
git clone git@github.com:yourname/yourname.github.io to_folder
