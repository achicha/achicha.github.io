Настраиваем Ubuntu после запуска.

Обновление

sudo apt-get update && sudo apt-get upgrade
# обновить только 1 программу
sudo apt-get install --only-upgade program

Отключаем комерческий сервис в меню Dash

 gsettings set com.canonical.Unity.Lenses disabled-scopes "['more_suggestions-amazon.scope', 'more_suggestions-u1ms.scope', 'more_suggestions-populartracks.scope', 'music-musicstore.scope', 'more_suggestions-ebay.scope', 'more_suggestions-ubuntushop.scope', 'more_suggestions-skimlinks.scope']"

Разблокировать dpkg

Часто появляется ошибка: Unable to lock the administration directory (/var/lib/dpkg/) is another process using it

ps -A | grep apt-get
sudo kill -9 <process-id>
sudo kill -9 2345

Создание загрузочной флешки из образа диска

sudo dd if=/home/username/path/filename.iso of=/dev/sdb bs=16M
  1. ставить на форматированную fat32 флешку.
  2. включить галку - загрузочный диск.
  3. ставлю на sdb раздел, на sdb1 не получается.
  4. bs=16m сколько за раз записывается.

Apple slim keyboard исправляем кнопки F1-F12 в Ubuntu 14.04

#Правильное Alt отображение и Command
echo options hid_apple swap_opt_cmd=1 | sudo tee -a /etc/modprobe.d/hid_apple.conf
#Восстановить работу функции кнопок 
#0 = отключено: Позволяет отключить клавиши 'Fn "
#1 = fkeyslast: Функциональные клавиши имеют низкий приоритет.
#2 = fkeysfirst: Функциональные клавиши имеют приоритет
echo options hid_apple fnmode=2 | sudo tee -a /etc/modprobe.d/hid_apple.conf
#Добавьте копирование конфигурации загрузки файлов initramfs.
sudo update-initramfs -u -k all
sudo reboot

Настраиваем VPN между ubuntu и windows 7 и заходим через xfreerdp

sudo apt-add-repository ppa:remmina-ppa-team/remmina-next
sudo apt-get update
apt-get install vpnc
apt-get install xfreerdp

# вставляем свои сертификаты в /etc/vpnc/
# запуск vpnc
sudo vpnc my.conf
#создаем скрипт с нужными правами rdp.sh с параметрами
# для 1.03
xfreerdp --plugin cliprdr -x 0x80 --disable-wallpaper --disable-full-window-drag --disable-menu-animations -g 1856x1027 -u my.login -p my.password -z --ignore-certificate my.machine.in.domain
# для 1.2
xfreerdp +clipboard /cert-ignore /u:login /p:password /v:my.machine.in.domain /size:1856x1027 -themes -wallpaper +fonts
#бывает отваливается и нужно перезагрузить
ps axu | grep vpnc
