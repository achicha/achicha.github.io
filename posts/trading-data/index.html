Где скачать исторические данные для акций?

  1. Finam.ru
  2. Google Finance
  3. Yahoo Finance
  4. Quandl.com

Подготовка

# для установки плагинов в Pycharm
sudo apt-get install python3-dev
# установка openpyxl, pandas, numpy

Finam

скачиваем с гитхаба файлик

mkdir finam
git clone git@github.com:achicha/finamdownloader finam

# создаем файл 1.py и добавляем туда

# коды бумаг, как в финам 
code = 'SiH5, RIH5'.replace(" ", "").split(',')
# даты начала и конца, исторических данных
start ='20141208'
end = '20141209'

code_lst = []
for i in code:
    if len(i) > 2:
        code_lst.append(i)
per = 'tick'

for y in code_lst:
    print('download %s data for %s' % (per, y))
    #скачиваем тики (можно и минутки качать)
    quote = get_quotes_finam(y, start_date=start, end_date= end, period=per)
    print(quote.head(n=3))

    url = '{0}.xlsx'.format(y+"_"+start+"_"+end)
    #сохраняем в excel
    with ExcelWriter(url) as writer:
        quote.to_excel(writer, y)
        #quote.to_excel(writer, 'Data 1')     #write to the second list
        print(y + ' saved to file')

Google Finance и Yahoo Finance

Исторические данные из этих источников, можно легко достать через встроенную в pandas функцию.

import pandas.io.data as web
import datetime

start = datetime.datetime(2011, 12, 1)
end = datetime.datetime(2014, 12, 18)

y = web.DataReader("AAPL", 'yahoo', start, end)
g = web.DataReader("AAPL", 'google', start, end)

print(y.tail(n=20))
print (g.tail(n=20)

Quandl

скачиваем с гитхаба библиотеку квандла для работы

import Quandl
from pandas import DataFrame, read_csv, ExcelWriter
#search for dataset
#datasets = Quandl.search(query = "euro bund", page = 2, prints = True)
def get_quotes_quandl(symbol, start_date, end_date):

    print('downloading data for %s' % (symbol))
    data = Quandl.get(symbol, authtoken='your_token', trim_start=start_date,
                    trim_end=end_date, collapse='daily', returns='pandas') 
                    # returns = "numpy" or "pandas"
    print(data.head(n=3))
    url = '{0}.xlsx'.format(symbol.replace("/", "."))
    with ExcelWriter(url) as writer:
        data.to_excel(writer, 'sheet1')
        #quote.to_excel(writer, 'Data 1')     #write to the second list
        print('saved to file '+url)
